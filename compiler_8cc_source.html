106 void Compiler::exitCond_compare(ModelicaParser::Cond_compareContext *ctx) {
107  auto left = cloneAst<ast::expression::Base>(ctx->expr(0));
108  auto right = cloneAst<ast::expression::Base>(ctx->expr(1));
109  std::string op = ctx->op->getText();
110  if (op == "<") {
111  setAst(ctx, std::make_unique<ast::condition::LessThan>(std::move(left),
112  std::move(right)));
113  } else if (op == "<=") {
114  setAst(ctx, std::make_unique<ast::condition::LessThanOrEqual>(
115  std::move(left), std::move(right)));
116  } else if (op == ">") {
117  setAst(ctx, std::make_unique<ast::condition::GreaterThan>(
118  std::move(left), std::move(right)));
119  } else if (op == ">=") {
120  setAst(ctx, std::make_unique<ast::condition::GreaterThanOrEqual>(
121  std::move(left), std::move(right)));
122  } else if (op == "==") {
123  setAst(ctx, std::make_unique<ast::condition::Equal>(std::move(left),
124  std::move(right)));
125  } else if (op == "<>") {
126  setAst(ctx, std::make_unique<ast::condition::NotEqual>(std::move(left),
127  std::move(right)));
128  } else {
129  throw compiler_exception(std::string("unhandled operator") + op);
130  }
