329 void Compiler::exitComposition(ModelicaParser::CompositionContext *ctx) {
330  auto c = std::make_unique<ast::model::Class>(
331  std::make_unique<ast::model::ElementDict>(),
332  std::make_unique<ast::equation::List>());
333 
334  // elements
335  enum class Visibility { PUBLIC, PRIVATE, PROTECTED };
336  for (auto elem_sect :
337  {std::make_pair(Visibility::PUBLIC, ctx->public_elem),
338  std::make_pair(Visibility::PRIVATE, ctx->private_elem),
339  std::make_pair(Visibility::PROTECTED, ctx->protected_elem)}) {
340  for (auto e_ctx : elem_sect.second) {
341  // TODO set visibility for element
342  auto edict = getAst<ast::model::ElementDict>(e_ctx);
343  for (auto &key_val : edict->getMap()) {
344  auto e = key_val.second->cloneAs<ast::element::Base>();
345  c->getElements().set(key_val.first, std::move(e));
346  }
347  }
348  }
349 
350  // equations
351  for (auto eq_sec : ctx->equation_section()) {
352  for (auto eq : eq_sec->equation()) {
353  c->getEquations().append(cloneAst<ast::equation::Base>(eq));
354  }
355  }
356 
357  // statements
358  for (auto algo_sec : ctx->algorithm_section()) {
359  for (auto s_ctx : algo_sec->statement()) {
360  auto s = cloneAst<ast::statement::Base>(s_ctx);
361  c->getStatements().append(std::move(s));
362  }
363  }
364  m_root = c.get();
365  setAst(ctx, std::move(c));
